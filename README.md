## Install

```bash
npm install --save ditto-react-2
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'ditto-react-2'
import 'ditto-react-2/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
